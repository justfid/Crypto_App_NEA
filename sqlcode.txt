CREATE TABLE IF NOT EXISTS `User` (
	`userId` integer primary key NOT NULL UNIQUE,
	`username` TEXT NOT NULL,
	`password` TEXT NOT NULL,
	`wishlistId` INTEGER NOT NULL,
	`portfolioId` INTEGER NOT NULL,
	`fiatPresetsId` INTEGER NOT NULL,
	`notesId` INTEGER NOT NULL,
	`topCoinsId` INTEGER NOT NULL,
FOREIGN KEY(`wishlistId`) REFERENCES `Wishlist`(`id`),
FOREIGN KEY(`portfolioId`) REFERENCES `PortfolioData`(`id`),
FOREIGN KEY(`fiatPresetsId`) REFERENCES `FiatPresets`(`id`),
FOREIGN KEY(`notesId`) REFERENCES `NotesData`(`id`),
FOREIGN KEY(`topCoinsId`) REFERENCES `TopCoins`(`id`)
);
CREATE TABLE IF NOT EXISTS `PortfolioData` (
	`id` integer primary key NOT NULL UNIQUE,
	`userId` INTEGER NOT NULL,
FOREIGN KEY(`userId`) REFERENCES `User`(`userId`)
);
CREATE TABLE IF NOT EXISTS `Coin` (
	`id` integer primary key NOT NULL UNIQUE,
	`coinName` TEXT NOT NULL,
	`ticker` TEXT NOT NULL
);
CREATE TABLE IF NOT EXISTS `FiatPresets` (
	`id` integer primary key NOT NULL UNIQUE,
	`presetName` TEXT NOT NULL,
	`coin1Id` INTEGER NOT NULL,
	`coin2Id` INTEGER NOT NULL,
FOREIGN KEY(`coin1Id`) REFERENCES `Coin`(`id`),
FOREIGN KEY(`coin2Id`) REFERENCES `Coin`(`id`)
);
CREATE TABLE IF NOT EXISTS `Wishlist` (
	`id` integer primary key NOT NULL UNIQUE,
	`date` REAL NOT NULL
);
CREATE TABLE IF NOT EXISTS `TopCoins` (
	`id` integer primary key NOT NULL UNIQUE,
	`coinId` INTEGER NOT NULL,
FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)
);
CREATE TABLE IF NOT EXISTS `Badges` (
	`id` integer primary key NOT NULL UNIQUE,
	`name` TEXT NOT NULL,
	`description` TEXT NOT NULL
);
CREATE TABLE IF NOT EXISTS `Transactions` (
	`portfolioId` integer primary key NOT NULL UNIQUE,
	`coinId` INTEGER NOT NULL,
	`date` REAL NOT NULL,
	`value` INTEGER NOT NULL,
FOREIGN KEY(`portfolioId`) REFERENCES `PortfolioData`(`id`),
FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)
);
CREATE TABLE IF NOT EXISTS `portfolioBadges` (
	`portfolioid` INTEGER NOT NULL,
	`badgeId` INTEGER NOT NULL,
FOREIGN KEY(`portfolioid`) REFERENCES `PortfolioData`(`id`),
FOREIGN KEY(`badgeId`) REFERENCES `Badges`(`id`)
);
CREATE TABLE IF NOT EXISTS `WishlistCoin` (
	`wishlistId` INTEGER NOT NULL,
	`coinId` INTEGER NOT NULL,
FOREIGN KEY(`wishlistId`) REFERENCES `Wishlist`(`id`),
FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)
);
CREATE TABLE IF NOT EXISTS `NotesData` (
	`id` integer primary key NOT NULL UNIQUE,
	`content` TEXT NOT NULL
);
FOREIGN KEY(`wishlistId`) REFERENCES `Wishlist`(`id`)
FOREIGN KEY(`portfolioId`) REFERENCES `PortfolioData`(`id`)
FOREIGN KEY(`fiatPresetsId`) REFERENCES `FiatPresets`(`id`)
FOREIGN KEY(`notesId`) REFERENCES `NotesData`(`id`)
FOREIGN KEY(`topCoinsId`) REFERENCES `TopCoins`(`id`)
FOREIGN KEY(`userId`) REFERENCES `User`(`userId`)

FOREIGN KEY(`coin1Id`) REFERENCES `Coin`(`id`)
FOREIGN KEY(`coin2Id`) REFERENCES `Coin`(`id`)

FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)

FOREIGN KEY(`portfolioId`) REFERENCES `PortfolioData`(`id`)
FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)
FOREIGN KEY(`portfolioid`) REFERENCES `PortfolioData`(`id`)
FOREIGN KEY(`badgeId`) REFERENCES `Badges`(`id`)
FOREIGN KEY(`wishlistId`) REFERENCES `Wishlist`(`id`)
FOREIGN KEY(`coinId`) REFERENCES `Coin`(`id`)
